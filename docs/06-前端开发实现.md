```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;

public class SeatController : Controller
{
    private readonly AppDbContext _context;
    public SeatController(AppDbContext context) => _context = context;

    public async Task<IActionResult> List(int page = 1)
    {
        int pageSize = 10;
        var seats = await _context.Seats.OrderBy(s => s.Id)
            .Skip((page - 1) * pageSize).Take(pageSize).ToListAsync();
        return View(seats);
    }

    public async Task<IActionResult> Detail(int id)
    {
        var seat = await _context.Seats.FindAsync(id);
        if (seat == null) return NotFound();
        var reservedSlots = await _context.Reservations
            .Where(r => r.SeatId == id && r.ReserveDate == DateTime.Today && r.Status == 0)
            .Select(r => r.TimeSlot).ToListAsync();
        ViewBag.ReservedSlots = reservedSlots;
        return View(seat);
    }
}
```

### 5.3 ReservationController.cs

```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;

public class ReservationController : Controller
{
    private readonly AppDbContext _context;
    public ReservationController(AppDbContext context) => _context = context;

    private string GetCurrentUser() => HttpContext.Session.GetString("CurrentUserName") ?? "张三";

    [HttpGet("Reservation/Create/{seatId:int}")]
    public async Task<IActionResult> Create(int seatId)
    {
        var seat = await _context.Seats.FindAsync(seatId);
        if (seat == null || seat.Status != 0) return NotFound();
        ViewBag.Seat = seat;
        return View(new Reservation { SeatId = seatId, ReserveDate = DateTime.Today });
    }

    [HttpPost]
    public async Task<IActionResult> Create(Reservation model)
    {
        var seat = await _context.Seats.FindAsync(model.SeatId);
        if (seat == null || seat.Status != 0) return NotFound();
        model.UserName = GetCurrentUser();
        model.Status = 0;
        model.CreatedAt = DateTime.Now;

        var conflict = await _context.Reservations.AnyAsync(r =>
            r.SeatId == model.SeatId && r.ReserveDate == model.ReserveDate &&
            r.TimeSlot == model.TimeSlot && r.Status == 0);
        if (conflict) { ViewBag.Error = "该时段已被预约"; ViewBag.Seat = seat; return View(model); }

        _context.Reservations.Add(model);
        await _context.SaveChangesAsync();
        return RedirectToAction(nameof(My));
    }

    public async Task<IActionResult> My()
    {
        var reservations = await _context.Reservations.Include(r => r.Seat)
            .Where(r => r.UserName == GetCurrentUser() && (r.Status == 0 || r.Status == 2))
            .OrderByDescending(r => r.ReserveDate).ToListAsync();
        return View(reservations);
    }

    [HttpPost("Reservation/Cancel/{id:int}")]
    public async Task<IActionResult> Cancel(int id)
    {
        var reservation = await _context.Reservations.FindAsync(id);
        if (reservation != null && reservation.UserName == GetCurrentUser() && reservation.Status == 0)
        { reservation.Status = 2; await _context.SaveChangesAsync(); }
        return RedirectToAction(nameof(My));
    }

    [HttpPost]
    public IActionResult SwitchUser(string userName)
    {
        HttpContext.Session.SetString("CurrentUserName", userName);
        return RedirectToAction("List", "Seat");
    }
}
```

### 5.4 AdminController.cs

```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;

public class AdminController : Controller
{
    private readonly AppDbContext _context;
    public AdminController(AppDbContext context) => _context = context;

    private bool IsLoggedIn() => HttpContext.Session.GetString("AdminUsername") != null;

    [HttpGet]
    public IActionResult Login() => IsLoggedIn() ? RedirectToAction("SeatIndex") : View();

    [HttpPost]
    public IActionResult Login(string username, string password)
    {
        if (_context.AdminUsers.Any(a => a.Username == username && a.Password == ComputeMd5(password)))
        {
            HttpContext.Session.SetString("AdminUsername", username);
            return RedirectToAction("SeatIndex");
        }
        ViewBag.Error = "用户名或密码错误";
        return View();
    }

    public IActionResult Logout() { HttpContext.Session.Remove("AdminUsername"); return RedirectToAction("Login"); }

    public async Task<IActionResult> SeatIndex()
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        return View(await _context.Seats.OrderBy(s => s.Id).ToListAsync());
    }

    [HttpGet]
    public IActionResult SeatCreate() => IsLoggedIn() ? View() : RedirectToAction("Login");

    [HttpPost]
    public async Task<IActionResult> SeatCreate(Seat seat)
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        if (await _context.Seats.AnyAsync(s => s.Name == seat.Name)) { ViewBag.Error = "座位编号已存在"; return View(seat); }
        seat.CreatedAt = DateTime.Now;
        _context.Seats.Add(seat);
        await _context.SaveChangesAsync();
        return RedirectToAction("SeatIndex");
    }

    [HttpGet]
    public async Task<IActionResult> SeatEdit(int id) => IsLoggedIn() ? View(await _context.Seats.FindAsync(id)) : RedirectToAction("Login");

    [HttpPost]
    public async Task<IActionResult> SeatEdit(Seat seat)
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        var existing = await _context.Seats.FindAsync(seat.Id);
        if (existing == null) return NotFound();
        existing.Name = seat.Name; existing.Location = seat.Location;
        existing.HasPower = seat.HasPower; existing.Status = seat.Status;
        await _context.SaveChangesAsync();
        return RedirectToAction("SeatIndex");
    }

    [HttpPost]
    public async Task<IActionResult> SeatDelete(int id)
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        var seat = await _context.Seats.FindAsync(id);
        if (seat != null && !await _context.Reservations.AnyAsync(r => r.SeatId == id))
        { _context.Seats.Remove(seat); await _context.SaveChangesAsync(); }
        return RedirectToAction("SeatIndex");
    }

    public async Task<IActionResult> ReservationIndex(DateTime? date, int? status)
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        var query = _context.Reservations.Include(r => r.Seat).AsQueryable();
        if (date.HasValue) query = query.Where(r => r.ReserveDate == date.Value);
        if (status.HasValue) query = query.Where(r => r.Status == status.Value);
        ViewBag.FilterDate = date?.ToString("yyyy-MM-dd");
        ViewBag.FilterStatus = status;
        return View(await query.OrderByDescending(r => r.ReserveDate).ToListAsync());
    }

    public async Task<IActionResult> Statistics()
    {
        if (!IsLoggedIn()) return RedirectToAction("Login");
        var today = DateTime.Today;
        ViewBag.TotalCount = await _context.Reservations.CountAsync();
        ViewBag.TodayCount = await _context.Reservations.CountAsync(r => r.ReserveDate == today && r.Status != 2);
        ViewBag.ActiveCount = await _context.Reservations.CountAsync(r => r.Status == 0);
        ViewBag.TopSeats = await _context.Reservations.GroupBy(r => r.SeatId)
            .Select(g => new { SeatId = g.Key, Count = g.Count() }).OrderByDescending(x => x.Count).Take(5)
            .Join(_context.Seats, x => x.SeatId, s => s.Id, (x, s) => new { s.Name, x.Count }).ToListAsync();
        var total = await _context.Reservations.CountAsync(r => r.Status != 2);
        ViewBag.TotalForPercentage = total;
        ViewBag.TimeSlotDistribution = await _context.Reservations.Where(r => r.Status != 2)
            .GroupBy(r => r.TimeSlot).Select(g => new { Slot = g.Key, Count = g.Count() })
            .OrderByDescending(x => x.Count).ToListAsync();
        return View();
    }

    private static string ComputeMd5(string input)
    {
        using var md5 = System.Security.Cryptography.MD5.Create();
        return BitConverter.ToString(md5.ComputeHash(System.Text.Encoding.UTF8.GetBytes(input))).Replace("-", "").ToLowerInvariant();
    }
```
